#include <iostream>
#include <cmath>

using namespace std;

int main() {
  float a1,b1,c1,a2,b2,c2,h;
    
  cout << "a1 = ";
  cin  >> a1;
    
  cout << "b1 = ";
  cin  >> b1;
    
  cout << "c1 = ";
  cin  >> c1;
    
  cout << "a2 = ";
  cin  >> a2;
  
  cout << "b2 = ";
  cin  >> b2;
  
  cout << "c2 = ";
  cin  >> c2;
  
  cout << "h  = ";
  cin  >> h;
  
  float p1=(a1+b1+c1)/2;
  float p2=(a2+b2+c2)/2;
  
  float s1=pow(p1*(p1-a1)*(p1-b1)*(p1-c1),0.5);
  float s2=pow(p2*(p2-a2)*(p2-b2)*(p2-c2),0.5);
  
  float ksi=h/4*(s1+2*pow(s1*s2,0.5)+3*s2)/(s1+pow(s1*s2,0.5)+s2);
  
  cout << "ksi = " << ksi/100;
  
  return 0;
}
